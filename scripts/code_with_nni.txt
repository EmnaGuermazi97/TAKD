# with nni
# launch a command line
nnictl create --config config.yml
or
nnictl create --config config.yml --port 8081
#if there is an error we use --debug
nnictl create --config config.yml --port 8081 --debug
# we find the path of the error log at the summary of the experiment: trials/stderr
# how to kill port
kill -9 ` lsof -t -i:8080`
